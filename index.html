function toReadable (number) {
    let first = {
        0: "zero",
        1: "one",
        2: "two",
        3: "three",
        4: "four",
        5: "five",
        6: "six",
        7: "seven",
        8: "eight",
        9: "nine"
    };

    let second = {
        10: "ten",
        11: "eleven",
        12: "twelve",
        13: "thirteen",
        14: "fourteen",
        15: "fifteen",
        16: "sixteen",
        17: "seventeen",
        18: "eighteen",
        19: "nineteen"
    };

    let third = {
        2: "twenty",
        3: "thirty",
        4: "forty",
        5: "fifty",
        6: "sixty",
        7: "seventy",
        8: "eighty",
        9: "ninety"
    };

    if (number < 10){
        return first[number];
    }else if(number < 100){
        let split2Num = number.toString().split('');
        if(number < 20){
            return second[number];
        } else{
            if(split2Num[1] === '0'){
                return third[split2Num[0]];
            } else{
                return `${third[split2Num[0]]} ${first[split2Num[1]]}`
            }
        }
    } else{
        let split3Num = number.toString().split('');
        if(number % 100 === 0){
            return `${first[split3Num[0]]} hundred`
        } else if(split3Num[1] === '0'){
            return `${first[split3Num[0]]} hundred ${first[split3Num[2]]}`
        } else if(number % 10 === 0 && split3Num[1] !== '1'){
            return `${first[split3Num[0]]} hundred ${third[split3Num[1]]}`
        } else if(split3Num[1] === '1'){
            return `${first[split3Num[0]]} hundred ${second[split3Num[1]+split3Num[2]]}`
        } else{
            return `${first[split3Num[0]]} hundred ${third[split3Num[1]]} ${first[split3Num[2]]}`
        }
    }
}
